from descriptifs_chantiers d
join localisation l using (id_chantier)
where st_dwithin(
	l.geom,
	st_setsrid(st_makepoint(2.3522, 48.8566), 4326),
	500
);

Calculer la distance entre deux chantiers :

select
	st_distance(
		l1.geom::geography,
		l2.geom::geography
	) as distance_m
from localisation l1
join localisation l2
on l1.id_chantier = 10
and l2.id_chantier = 25;

VUE 1 : Géolocalisations des chantiers

create view v_chantiers_geo as
select
	d.id_chantier,
	d.nature_operation,
	d.responsable,
	d.date_operation,
	d.synthese,
	d.periode,
	l.adresse,
	l.code postal
	l.latitude,
	l.longitude,
	l.geom
from descriptifs_chantiers d
join localisation l using (id_chantier);

VUE 2 : Répartition des chantiers par type d'opération

create view v_chantiers_par_type as
select
	nature_operation,
	count(*) as nb_chantiers
from descriptifs_chantiers
group by nature_operation
order by nb_chantiers desc;

VUE 3 : Chantiers classés par années

create view v_chantiers_par_annee as
select
	extract(Year from date_operation) as annee,
	count(*) as nb_chantiers
from descriptifs_chantiers
group by annee
order by annee;

VUE 4 : Chantiers avec géométrie valide uniquement

create view v_chantiers_geoçvalides as
select *
from v_chantiers_geo
where geom is not null and st_isvalid(geom);

VUE 5 : Chantiers dans Paris intra-muros (ne comprends pas la Petite ni la Grande Couronne)

-> utilisation de la bounding box de Paris

create view v_chantiers_paris as
select *
from v_chantiers_geo
where geom && st_makeenvelope(
	2.224, 48.815,
	2.469, 48.902,
	4326
);

VUE 6 : Chantiers proches les uns des autres

create view v_chantiers_proximite as
select
	a.id_chantier as chantier1,
	b.id_chantier as chantier2,
	st_distance(a.geom::geography, b.geom::geography) as distance_m
from localisation a
from localisation b
on a.id_chantier < b.id_chantier
and st_dwithin(a.geom, b.geom, 200);

VUE 7 : Chantiers avec buffer (zones d'influence)

create view v_chantiers_buffer_200m as
select
	id_chantier,
	geom,
	st_buffer(geom::geography, 200)::geometry as zone_200m
from localisation;

VUE 8 : Vue catalogue (vue de toutes les métadonnées essentielles)

create view v_chantiers_catalogue as
select
	d.id_chantier,
	d.nature_operation,
	d.date_operation,
	d.responsable,
	l.adresse,
	l.code_postal,
	l.latitude,
	l.longitude,
from descriptifs_chantiers d
join localisation l using (id_chantier)
order by d.date_operation desc;
